function saveDeal(event) { event.preventDefault(); let client = document.getElementById("client").value; let freelancer = document.getElementById("freelancer").value; let work = document.getElementById("work").value; let amount = document.getElementById("amount").value; let deal = { client: client, freelancer: freelancer, work: work, amount: amount }; // Save in local storage localStorage.setItem("deal", JSON.stringify(deal)); displayDeal(); } function displayDeal() { let savedDeal = JSON.parse(localStorage.getItem("deal")); if (savedDeal) { let result = "Client: " + savedDeal.client + "
" + "Freelancer: " + savedDeal.freelancer + "
" + "Work: " + savedDeal.work + "
" + "Amount: " + savedDeal.amount; document.getElementById("output").innerHTML = result; } } // Load saved data when page opens window.onload = displayDeal;
